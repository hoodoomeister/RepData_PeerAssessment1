data["weekdays"] <- NA
data$weekdays <- weekdays(as.Date(data[, 2]))

weekday <- c("Monday", "Tuesday", "Wednesday", "Thursday", "Friday")
weekend <- c("Saturday", "Sunday")
weekend <- as.matrix(weekend)
weekday <- as.matrix(weekday)

for (i in range1) {
    temp1 <- data[i, 4]
    temp1 <- as.character(temp1)
    if (temp1 %in% weekday) {
        data[i, 4] <- "Weekday"
    }
    if (temp1 %in% weekend) {
        data[i, 4] <- "Weekend"
    }
}
